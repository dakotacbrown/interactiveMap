function openLocation(evt, locationName) {
  var i, x, tablinks;
  x = document.getElementsByClassName("locations");
  for (i = 0; i < x.length; i++) {
     x[i].style.display = "none";
  }
  tablinks = document.getElementsByClassName("tablink");
  for (i = 0; i < x.length; i++) {
      tablinks[i].className = tablinks[i].className.replace(" w3-red", "");
  }
  document.getElementById(locationName).style.display = "block";
  evt.currentTarget.className += " w3-red";
}

function openTab(evt, hourName) {
  var i, x, tablonks;
  x = document.getElementsByClassName("hours");
  for (i = 0; i < x.length; i++) {
      x[i].style.display = "none";
  }
  tablonks = document.getElementsByClassName("tablonk");
  for (i = 0; i < x.length; i++) {
      tablonks[i].className = tablonks[i].className.replace(" w3-red", "");
  }
  document.getElementById(hourName).style.display = "block";
  evt.currentTarget.className += " w3-red";
}
